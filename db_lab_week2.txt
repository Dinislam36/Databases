CREATE TABLE production_company(
    name VARCHAR(100) PRIMARY KEY,
    address VARCHAR(40)
);

CREATE TABLE movie(
    title VARCHAR(100) UNIQUE ,
    year INT,
    length INT,
    production_company VARCHAR(100),
    plot_outline VARCHAR(400),
    PRIMARY KEY (title, year),
    FOREIGN KEY (production_company) REFERENCES production_company(name)
);

CREATE TABLE actor(
    actor_name VARCHAR(40) UNIQUE,
    actor_birth INT,
    PRIMARY KEY (actor_name, actor_birth)
);

CREATE TABLE director(
    director_name VARCHAR(40) UNIQUE ,
    director_birth INT,
    PRIMARY KEY (director_name, director_birth)
);

CREATE TABLE genres(
    title VARCHAR(40),
    genre VARCHAR(40),
    PRIMARY KEY (title, genre),
    FOREIGN KEY (title) REFERENCES movie(title)
);

CREATE TABLE quotes(
    title VARCHAR(40),
    actor VARCHAR(40),
    quote VARCHAR(100),
    PRIMARY KEY (title, actor),
    FOREIGN KEY (title) REFERENCES movie(title),
    FOREIGN KEY (actor) REFERENCES actor(actor_name)
);

CREATE TABLE actor_cast(
    title VARCHAR(40),
    actor VARCHAR(40),
    role VARCHAR(40),
    PRIMARY KEY (title, actor),
    FOREIGN KEY (title) REFERENCES movie(title),
    FOREIGN KEY (actor) REFERENCES actor(actor_name)
);

CREATE TABLE director_cast(
    title VARCHAR(40),
    director VARCHAR(40),
    PRIMARY KEY (title, director),
    FOREIGN KEY (title) REFERENCES movie(title),
    FOREIGN KEY (director) REFERENCES director(director_name)
);
